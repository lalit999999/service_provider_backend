  -H "Content-Type: application/json" \
  -d '{
    "name": "John",
    "email": "john@example.com",
    "password": "pass123",
    "role": "customer",
    "city": "New York",
    "area": "Manhattan"
  }'

  -H "Content-Type: application/json" \
