# Register
curl -X POST http://localhost:3000/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John",
    "email": "john@example.com",
    "password": "pass123",
    "role": "customer",
    "city": "New York",
    "area": "Manhattan"
  }'

# Login
curl -X POST http://localhost:3000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "john@example.com",
    "password": "pass123"
  }'
# logout
curl -X POST http://localhost:3000/api/auth/logout \
  -H "Content-Type: application/json"


# Get all services
curl -X GET http://localhost:3000/api/services

# Filter by category
curl -X GET "http://localhost:3000/api/services?category=plumbing"

# Filter by city
curl -X GET "http://localhost:3000/api/services?city=Patna"

# Filter by both category and city
curl -X GET "http://localhost:3000/api/services?category=plumbing&city=Patna"

# Get single service
curl -X GET http://localhost:3000/api/services/SERVICE_ID

# Create service (provider token required)
TOKEN="your_provider_jwt_token"
CATEGORY_ID="existing_category_id"

curl -X POST http://localhost:3000/api/services \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "categoryId": "'$CATEGORY_ID'",
    "title": "Professional Plumbing",
    "description": "Expert plumbing repairs and installations",
    "basePrice": 500
  }'

# Update service (provider only)
curl -X PUT http://localhost:3000/api/services/SERVICE_ID \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "Premium Plumbing Services",
    "description": "High-quality plumbing solutions",
    "basePrice": 600
  }'

# Delete service (provider only)
curl -X DELETE http://localhost:3000/api/services/SERVICE_ID \
  -H "Authorization: Bearer $TOKEN"


  # Create booking (customer token required)
TOKEN="customer_jwt_token"
SERVICE_ID="service_id"

curl -X POST http://localhost:3000/api/bookings \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "serviceId": "'$SERVICE_ID'",
    "address": "123 Main Street, Patna",
    "dateTime": "2026-03-15T10:00:00Z",
    "notes": "Please call before arrival",
    "priceAtBooking": 500
  }'

# Get my bookings
curl -X GET http://localhost:3000/api/bookings \
  -H "Authorization: Bearer $TOKEN"

# Get single booking
curl -X GET http://localhost:3000/api/bookings/BOOKING_ID \
  -H "Authorization: Bearer $TOKEN"

# Provider accepts booking
PROVIDER_TOKEN="provider_jwt_token"
curl -X PUT http://localhost:3000/api/bookings/BOOKING_ID/status \
  -H "Authorization: Bearer $PROVIDER_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"status": "Accepted"}'

# Provider marks booking completed
curl -X PUT http://localhost:3000/api/bookings/BOOKING_ID/status \
  -H "Authorization: Bearer $PROVIDER_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"status": "Completed"}'

# Cancel booking
curl -X PUT http://localhost:3000/api/bookings/BOOKING_ID/cancel \
  -H "Authorization: Bearer $TOKEN"
